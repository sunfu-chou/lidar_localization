302 void LidarLocalization::getRobotPose()
303 {
304  if (!validateBeaconGeometry())
305  {
306  ROS_WARN_STREAM("geometry error");
307  return;
308  }
309 
310  vector<double> dist_beacon_robot;
311  for (int i = 0; i < 3; ++i)
312  {
313  dist_beacon_robot.push_back(length(beacon_found_[i]));
314  }
315 
316  // least squares method to solve Ax=b
317  // i.e to solve (A^T)Ax=(A^T)b
318  mat A(2, 2);
319  vec b(2);
320  vec X(2);
321 
322  A(0, 0) = 2 * (beacon_to_map_[0].x - beacon_to_map_[2].x);
323  A(0, 1) = 2 * (beacon_to_map_[0].y - beacon_to_map_[2].y);
324 
325  A(1, 0) = 2 * (beacon_to_map_[1].x - beacon_to_map_[2].x);
326  A(1, 1) = 2 * (beacon_to_map_[1].y - beacon_to_map_[2].y);
327 
328  b(0) = (pow(beacon_to_map_[0].x, 2) - pow(beacon_to_map_[2].x, 2)) +
329  (pow(beacon_to_map_[0].y, 2) - pow(beacon_to_map_[2].y, 2)) +
330  (pow(dist_beacon_robot[2], 2) - pow(dist_beacon_robot[0], 2));
331  b(1) = (pow(beacon_to_map_[1].x, 2) - pow(beacon_to_map_[2].x, 2)) +
332  (pow(beacon_to_map_[1].y, 2) - pow(beacon_to_map_[2].y, 2)) +
333  (pow(dist_beacon_robot[2], 2) - pow(dist_beacon_robot[1], 2));
334  try
335  {
336  X = solve(A.t() * A, A.t() * b, solve_opts::no_approx);
337 
338  output_robot_pose_.pose.pose.position.x = X(0);
339  output_robot_pose_.pose.pose.position.y = X(1);
340 
341  double robot_yaw = 0;
342  double robot_sin = 0;
343  double robot_cos = 0;
344 
345  for (int i = 0; i < 3; i++)
346  {
347  double theta = atan2(beacon_to_map_[i].y - output_robot_pose_.pose.pose.position.y,
348  beacon_to_map_[i].x - output_robot_pose_.pose.pose.position.x) -
349  atan2(beacon_found_[i].y, beacon_found_[i].x);
350 
351  robot_sin += sin(theta);
352  robot_cos += cos(theta);
353  }
354 
355  robot_yaw = atan2(robot_sin, robot_cos);
356  tf2::Quaternion q;
357  q.setRPY(0., 0., robot_yaw);
358  output_robot_pose_.pose.pose.orientation = tf2::toMsg(q);
359 
360  publishLocation();
361  }
362  catch (const std::runtime_error& ex)
363  {
364  ROS_WARN_STREAM(A);
365  ROS_WARN_STREAM(b);
366  ROS_WARN_STREAM(ex.what());
367  }
368  publishBeacons();
369 }
